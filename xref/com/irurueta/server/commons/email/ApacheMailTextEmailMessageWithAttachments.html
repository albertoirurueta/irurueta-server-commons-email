16  package com.irurueta.server.commons.email;
17  
18  import java.util.List;
19  
20  import org.apache.commons.mail.EmailException;
21  import org.apache.commons.mail.MultiPartEmail;
22  
23  /**
24   * Class containing a text only email message that can contain file attachments
25   * and is meant to be sent using Apache Mail.
26   */
27  public class ApacheMailTextEmailMessageWithAttachments extends TextEmailMessageWithAttachments
28          implements ApacheMultipartEmailMessage {
29  
30      /**
31       * Constructor.
32       */
33      public ApacheMailTextEmailMessageWithAttachments() {
34          super();
35      }
36  
37      /**
38       * Constructor with email subject.
39       *
40       * @param subject subject to be set on email.
41       */
42      public ApacheMailTextEmailMessageWithAttachments(final String subject) {
43          super(subject);
44      }
45  
46      /**
47       * Constructor with email subject and content text.
48       *
49       * @param subject subject to be set on email.
50       * @param text    textual email content.
51       */
52      public ApacheMailTextEmailMessageWithAttachments(final String subject,
53                                                       final String text) {
54          super(subject, text);
55      }
56  
57      /**
58       * Constructor with file attachments.
59       *
60       * @param attachments list of file attachments to be included in email.
61       */
62      public ApacheMailTextEmailMessageWithAttachments(
63              final List<EmailAttachment> attachments) {
64          super(attachments);
65      }
66  
67      /**
68       * Constructor with subject and file attachments.
69       *
70       * @param subject     subject to be set on email.
71       * @param attachments list of file attachments to be included in email.
72       */
73      public ApacheMailTextEmailMessageWithAttachments(final String subject,
74                                                       final List<EmailAttachment> attachments) {
75          super(subject, attachments);
76      }
77  
78      /**
79       * Constructor with subject, content text and file attachments.
80       *
81       * @param subject     subject to be set on email.
82       * @param text        textual email content.
83       * @param attachments list of file attachments to be included in email.
84       */
85      public ApacheMailTextEmailMessageWithAttachments(final String subject,
86                                                       final String text, final List<EmailAttachment> attachments) {
87          super(subject, text, attachments);
88      }
89  
90      /**
91       * Builds email content to be sent using an email sender.
92       *
93       * @param content instance where content must be set.
94       * @throws com.irurueta.server.commons.email.EmailException if setting mail
95       *                                                          content fails.
96       */
97      @Override
98      public void buildContent(final MultiPartEmail content)
99              throws com.irurueta.server.commons.email.EmailException {
100         try {
101             if (getText() != null) {
102                 content.setMsg(getText());
103             }
104 
105             // add attachments
106             final List<EmailAttachment> attachments = getAttachments();
107             attach(content, attachments);
108         } catch (final EmailException e) {
109             throw new com.irurueta.server.commons.email.EmailException(e);
110         }
111     }
112 
113     /**
114      * Adds attachments to email content.
115      *
116      * @param content a multipart email.
117      * @param attachments attachments to be added.
118      * @throws EmailException if an error occurs.
119      */
120     protected static void attach(final MultiPartEmail content, final List<EmailAttachment> attachments)
121             throws EmailException {
122         org.apache.commons.mail.EmailAttachment apacheAttachment;
123         if (attachments != null) {
124             for (final EmailAttachment attachment : attachments) {
125                 // only add attachments with files
126                 if (attachment.getAttachment() == null) {
127                     continue;
128                 }
129 
130                 apacheAttachment = new org.apache.commons.mail.
131                         EmailAttachment();
132                 apacheAttachment.setPath(attachment.getAttachment().getAbsolutePath());
133                 apacheAttachment.setDisposition(
134                         org.apache.commons.mail.EmailAttachment.ATTACHMENT);
135                 if (attachment.getName() != null) {
136                     apacheAttachment.setName(attachment.getName());
137                 }
138 
139                 content.attach(apacheAttachment);
140             }
141         }
142     }
143 }
