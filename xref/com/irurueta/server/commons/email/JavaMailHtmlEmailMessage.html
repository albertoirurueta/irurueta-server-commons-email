16  package com.irurueta.server.commons.email;
17  
18  import java.text.MessageFormat;
19  import java.util.ArrayList;
20  import java.util.List;
21  import javax.activation.DataHandler;
22  import javax.activation.FileDataSource;
23  import javax.mail.BodyPart;
24  import javax.mail.MessagingException;
25  import javax.mail.Multipart;
26  import javax.mail.Part;
27  import javax.mail.internet.MimeBodyPart;
28  import javax.mail.internet.MimeMessage;
29  import javax.mail.internet.MimeMultipart;
30  
31  /**
32   * Class to build emails having HTML markup as in webpages using Java Mail.
33   * These kind of emails are useful for marketing purposes or simply to get a
34   * nicer layout.
35   * Files can be attached as in text emails, or they can be sent inline so they
36   * can be used within the html content (such as images or css resources to be
37   * used within html email body).
38   */
39  public class JavaMailHtmlEmailMessage extends HtmlEmailMessage<MimeMessage> {
40  
41      /**
42       * Constructor.
43       */
44      public JavaMailHtmlEmailMessage() {
45          super();
46      }
47      
48      /**
49       * Constructor with email subject.
50       * @param subject subject to be set.
51       */    
52      public JavaMailHtmlEmailMessage(String subject) {
53          super(subject);
54      }
55      
56      /**
57       * Constructor with email subject and textual content.
58       * @param subject subject to be set.
59       * @param htmlContent HTML content.
60       */    
61      public JavaMailHtmlEmailMessage(String subject, String htmlContent) {
62          super(subject, htmlContent);
63      }
64      
65      /**
66       * Builds email content to be sent using an email sender.
67       * @param message instance where content must be set.
68       * @throws EmailException if setting mail content fails.
69       */    
70      @Override
71      protected void buildContent(MimeMessage message) throws EmailException {
72          try {
73              
74              //finally add multipart contents to mail message
75              message.setContent(buildMultipart());                        
76          } catch (MessagingException e) {
77              throw new EmailException(e);
78          }
79      }  
80      
81      /**
82       * Builds internal multipart email content.
83       * @return a multipart message.
84       * @throws EmailException if something fails.
85       */
86      private Multipart buildMultipart() throws EmailException {
87          try {
88              //create multipart. One part will be for alternative text content, 
89              //another for HTML content and remaining parts will be for inline 
90              //attachments and other attachments
91              Multipart multipart = new MimeMultipart("related");
92              BodyPart messageBodyPart;
93                          
94              if (getAlternativeText() != null || getHtmlContent() != null) {
95                  //prepare alternative multipart so it can hold both alternative
96                  //text and html
97                  Multipart alternativeMultipart = new MimeMultipart(
98                          "alternative");
99                  if (getAlternativeText() != null) {
100                     messageBodyPart = new MimeBodyPart();
101                     messageBodyPart.setContent(getAlternativeText(), 
102                             "text/plain; charset=utf-8");                
103                     alternativeMultipart.addBodyPart(messageBodyPart);
104                 }
105 
106                 //process inline attachments
107                 boolean[] generated = processInlineAttachments();
108 
109                 //set html content in another body part
110                 if (getHtmlContent() != null) {
111                     messageBodyPart = new MimeBodyPart();
112                     messageBodyPart.setContent(process(getHtmlContent(), generated), 
113                             "text/html; charset=utf-8");
114                     alternativeMultipart.addBodyPart(messageBodyPart);
115                 }
116                 
117                 //wrap alternative multipart in a body part to be included in
118                 //overall message multipart
119                 messageBodyPart = new MimeBodyPart();
120                 messageBodyPart.setContent(alternativeMultipart);
121                 multipart.addBodyPart(messageBodyPart);
122             }
123             
124             //all attachments go into general message multipar
125             
126             //add inline attachments
127             List<InlineAttachment> inlineAttachments = getInlineAttachments();
128             if (inlineAttachments != null) {
129                 for (InlineAttachment attachment : inlineAttachments) {
130                     //only add attachments with files and content ids
131                     if (attachment.getContentId() == null || 
132                             attachment.getAttachment() == null) {
133                         continue;
134                     }
135                     
136                     messageBodyPart = new MimeBodyPart();
137                     messageBodyPart.setDataHandler(new DataHandler(
138                             new FileDataSource(attachment.getAttachment())));
139                     messageBodyPart.setHeader("Content-ID", 
140                             "<" + attachment.getContentId() + ">");
141                     if (attachment.getContentType() != null) {
142                         messageBodyPart.addHeader("Content-Type", 
143                                 attachment.getContentType());
144                     }
145                     messageBodyPart.setDisposition(Part.INLINE);
146                     multipart.addBodyPart(messageBodyPart);                    
147                 }
148             }
149             
150             //add other attachments parts
151             List<EmailAttachment> attachments = getEmailAttachments();
152             if (attachments != null) {
153                 for (EmailAttachment attachment : attachments) {
154                     //only add attachments with files
155                     if (attachment.getAttachment() == null) {
156                         continue;
157                     }
158 
159                     messageBodyPart = new MimeBodyPart();
160                     if (attachment.getName() != null) {
161                         messageBodyPart.setFileName(attachment.getName());
162                     }
163                     messageBodyPart.setDisposition(Part.ATTACHMENT);
164                     if (attachment.getContentType() != null) {
165                         messageBodyPart.addHeader("Content-Type", 
166                                 attachment.getContentType());
167                     }
168                     messageBodyPart.setDataHandler(new DataHandler(
169                             new FileDataSource(attachment.getAttachment())));
170                     multipart.addBodyPart(messageBodyPart);
171                 }
172             }
173             return multipart;
174         } catch (MessagingException e) {
175             throw new EmailException(e);
176         }                        
177     }   
178     
179     /**
180      * Reads files to be attached inline in HTML content.
181      * @return array indicating for whith files content was inlined.
182      */
183     private boolean[] processInlineAttachments() {
184         List<InlineAttachment> attachments = getInlineAttachments();
185         if (attachments != null) {
186             boolean[] result = new boolean[attachments.size()];
187             int counter = 0;
188             boolean isGeneratedId;
189             String contentId;
190             for (InlineAttachment attachment : attachments) {
191                 isGeneratedId = attachment.getContentId() == null;
192                 result[counter] = isGeneratedId;
193                 if (isGeneratedId) {
194                     //generate a content id for this attachment
195                     contentId = "inline-attachment" + counter;
196                     attachment.setContentId(contentId);
197                 }                
198                 counter++;
199             }
200             return result;
201         }
202         return null;
203     }
204     
205     /**
206      * Processes HTML content to substitute placeholders by their corresponding
207      * //inline attachment ids.
208      * @param htmlContent HTML content to be sent.
209      * @param generated array containing inlined files that where found as 
210      * placeholders and correctly inlined into content.
211      * @return resulting content.
212      */
213     private String process(String htmlContent, boolean[] generated) {
214         //if no information about generated inline ids is available, then simply
215         //return input html content
216         if (generated == null) {
217             return htmlContent;
218         }
219         
220         //process html content to substitute placeholders by their corresponding
221         //inline attachments ids
222         List<InlineAttachment> attachments = getInlineAttachments();
223         List<String> contentIds = new ArrayList<String>();
224         int pos = 0;
225         if (attachments != null) {
226             for (InlineAttachment attachment : attachments) {
227                 if (generated[pos]) {
228                     contentIds.add(attachment.getContentId());
229                 }
230                 pos++;
231             }
232         }
233         
234         Object[] objects = contentIds.toArray();
235         return MessageFormat.format(htmlContent, objects);
236     }
