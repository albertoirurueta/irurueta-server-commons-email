16  package com.irurueta.server.commons.email;
17  
18  import java.text.MessageFormat;
19  import java.util.ArrayList;
20  import java.util.List;
21  import org.apache.commons.mail.EmailException;
22  import org.apache.commons.mail.HtmlEmail;
23  
24  /**
25   * Class to build emails having HTML markup as in webpages using Apache Mail.
26   * These kind of emails are useful for marketing purposes or simply to get a
27   * nicer layout.
28   * Files can be attached as in text emails, or they can be sent inline so they
29   * can be used within the html content (such as images or css resources to be
30   * used within html email body).
31   */
32  public class ApacheMailHtmlEmailMessage extends HtmlEmailMessage<HtmlEmail> {
33  
34      /**
35       * Constructor.
36       */
37      public ApacheMailHtmlEmailMessage() {
38          super();
39      }
40      
41      /**
42       * Constructor with email subject.
43       * @param subject subject to be set.
44       */    
45      public ApacheMailHtmlEmailMessage(String subject) {
46          super(subject);
47      }
48      
49      /**
50       * Constructor with email subject and textual content.
51       * @param subject subject to be set.
52       * @param htmlContent HTML content.
53       */    
54      public ApacheMailHtmlEmailMessage(String subject, String htmlContent) {
55          super(subject, htmlContent);
56      }
57      
58      /**
59       * Builds email content to be sent using an email sender.
60       * @param content instance where content must be set.
61       * @throws com.irurueta.server.commons.email.EmailException if setting mail 
62       * content fails.
63       */        
64      @Override
65      protected void buildContent(HtmlEmail content) 
66              throws com.irurueta.server.commons.email.EmailException {
67          try {
68              if (getAlternativeText() != null) {
69                  content.setTextMsg(getAlternativeText());
70              }
71              
72              //process inline attachments
73              boolean[] generated = processInlineAttachments();
74              
75              if (getHtmlContent() != null) {
76                  content.setHtmlMsg(process(getHtmlContent(), generated));
77              }
78              
79              //all attachments go into general message multipar
80              
81              //add inline attachments
82              List<InlineAttachment> inlineAttachments = getInlineAttachments();
83              if (inlineAttachments != null) {
84                  for(InlineAttachment attachment : inlineAttachments){
85                      //only add attachments with files and content ids
86                      if (attachment.getContentId() == null || 
87                              attachment.getAttachment() == null) {
88                          continue;
89                      }
90                      
91                      content.embed(attachment.getAttachment(), 
92                              attachment.getContentId());
93                  }
94              }
95              
96              //add other attachments parts
97              List<EmailAttachment> attachments = getEmailAttachments();
98              org.apache.commons.mail.EmailAttachment apacheAttachment;
99              if (attachments != null) {
100                 for (EmailAttachment attachment : attachments) {
101                     //only add attachments with files
102                     if (attachment.getAttachment() == null) {
103                         continue;
104                     }
105 
106                     apacheAttachment = new  org.apache.commons.mail.
107                             EmailAttachment();
108                     apacheAttachment.setPath(attachment.getAttachment().getAbsolutePath());
109                     apacheAttachment.setDisposition(
110                             org.apache.commons.mail.EmailAttachment.ATTACHMENT);
111                     if (attachment.getName() != null) {
112                         apacheAttachment.setName(attachment.getName());
113                     }
114                     
115                     content.attach(apacheAttachment);
116                 }
117             }                        
118         } catch (EmailException e) {
119             throw new com.irurueta.server.commons.email.EmailException(e);
120         }
121     }
122     
123     /**
124      * Reads files to be attached inline in HTML content.
125      * @return array indicating for whith files content was inlined.
126      */
127     private boolean[] processInlineAttachments() {
128         List<InlineAttachment> attachments = getInlineAttachments();
129         if (attachments != null) {
130             boolean[] result = new boolean[attachments.size()];
131             int counter = 0;
132             boolean isGeneratedId;
133             String contentId;
134             for (InlineAttachment attachment : attachments ) {
135                 isGeneratedId = attachment.getContentId() == null;
136                 result[counter] = isGeneratedId;
137                 if (isGeneratedId) {
138                     //generate a content id for this attachment
139                     contentId = "inline-attachment" + counter;
140                     attachment.setContentId(contentId);
141                 }                
142                 counter++;
143             }
144             return result;
145         }
146         return null;
147     }
148     
149     /**
150      * Processes HTML content to substitute placeholders by their corresponding
151      * //inline attachment ids.
152      * @param htmlContent HTML content to be sent.
153      * @param generated array containing inlined files that where found as 
154      * placeholders and correctly inlined into content.
155      * @return resulting content.
156      */    
157     private String process(String htmlContent, boolean[] generated) {
158         //if no information about generated inline ids is available, then simply
159         //return input html content
160         if (generated == null) {
161             return htmlContent;
162         }
163         
164         //process html content to substitute placeholders by their corresponding
165         //inline attachments ids
166         List<InlineAttachment> attachments = getInlineAttachments();
167         List<String> contentIds = new ArrayList<>();
168         int pos = 0;
169         if (attachments != null) {
170             for (InlineAttachment attachment : attachments) {
171                 if (generated[pos]) {
172                     contentIds.add(attachment.getContentId());
173                 }
174                 pos++;
175             }
176         }
177         
178         Object[] objects = contentIds.toArray();
179         return MessageFormat.format(htmlContent, objects);
180     }
