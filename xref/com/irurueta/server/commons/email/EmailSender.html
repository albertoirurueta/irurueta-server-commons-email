16  package com.irurueta.server.commons.email;
17  
18  import com.irurueta.server.commons.configuration.ConfigurationException;
19  import java.util.regex.Matcher;
20  import java.util.regex.Pattern;
21  
22  /**
23   * Base class in charge of sending emails.
24   * This class is a singleton and an instance must be obtained using getInstance
25   * and using current mail configuration.
26   */
27  public abstract class EmailSender {
28  
29      /**
30       * Regular expression to validate emails.
31       */
32      private static final String EMAIL_REGEX = "^[\\w.-]+@([\\w.-]+\\.)+[a-z]{2,4}$";
33  
34      /**
35       * Factory method to obtain singleton instance using current mail
36       * configuration.
37       * @return singleton instance.
38       * @throws ConfigurationException if mail configuration fails.
39       */
40      public static EmailSender getInstance()
41              throws ConfigurationException {
42          
43          return getInstance(MailConfigurationFactory.getInstance().
44                  configure().getProvider());
45      }
46      
47      /**
48       * Factory method to obtain appropriate subclass for a given email provider.
49       * @param provider provider to use.
50       * @return an email sender controller singleton.
51       */
52      protected static EmailSender getInstance(final EmailProvider provider) {
53          switch (provider) {
54              case AWS_MAIL:
55                  return AWSMailSender.getInstance();
56              case APACHE_MAIL:
57                  return ApacheMailSender.getInstance();
58              case JAVA_MAIL:
59              default:
60                  return JavaMailSender.getInstance();
61          }
62      }
63  
64      /**
65       * Determines if provided string corresponds to an email address.
66       * This method checks proper format of provided string to match the 
67       * structure of an email address (i.e. user@host).
68       * @param email string containing an email address to be checked.
69       * @return true if provided string is a valid email address, false 
70       * otherwise.
71       */    
72      public static boolean isValidEmailAddress(final String email) {
73          /*
74          Email format: A valid email address will have following format:
75          [\\w\\.-]+: Begins with word characters, (may include periods and hypens).
76          @: It must have a '@' symbol after initial characters.
77          ([\\w\\.-]+\\.)+: '@' must follow by more alphanumeric characters (may 
78          * include hypens or dots to allow subdomains).
79          This part must also have a "." to separate domain and subdomain names.
80          [A-Z]{2,4}$ : Must end with two to four alphabets.
81          (This will allow domain names with 2, 3 and 4 characters e.g pa, com, net, wxyz)
82  
83          Examples: Following email addresses will pass validation
84          abc@xyz.net; ab.c@tx.gov
85          */
86  
87          // Initialize reg ex for email.
88          // Make the comparison case-insensitive.
89          final Pattern pattern = Pattern.compile(EMAIL_REGEX, Pattern.CASE_INSENSITIVE);
90          final Matcher matcher = pattern.matcher(email);
91          return matcher.matches();        
92      }
93      
94      /**
95       * Abstract method to send an email.
96       * @param m email message to be sent.
97       * @return id of message that has been sent.
98       * @throws MailNotSentException if mail couldn't be sent.
99       * @throws InterruptedException if thread is interrupted.
100      */
101     public abstract String send(final EmailMessage<?> m) throws MailNotSentException, InterruptedException;
102     
103     /**
104      * Returns provider used by this email sender.
105      * @return email provider.
106      */
107     public abstract EmailProvider getProvider();
108 }
